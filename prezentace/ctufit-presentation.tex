\documentclass[czech]{ctufit-presentation}

\usepackage[backend=biber]{biblatex}
\addbibresource{data.bib}

\ctufittitle{Název prezentace}
\ctufitsubtitle{Podnadpis prezentace} % Remove the command if you do not want a subtitle on the title slide.
\ctufitauthor{Author Authorovič} % Remove the command if you do not want the author on the title slide.
\ctufitdate{\today} % Remove the command if you do not want the date on the title slide.
\ctufitinstitute{FIT CVUT} % Remove the command if you do not want the institution on the title slide.

\begin{document}

\begin{frame}[plain,t]
	\titlepage
\end{frame}

\begin{frame}
	\frametitle{Nadpis}
	\tableofcontents
\end{frame}

%=============================================================================================

\section{Úvod}
\begin{frame}
	\frametitle{Slide}
	\framesubtitle{Bodový seznam}
	\begin{itemize}
		\item První bod
			\begin{itemize}
				\item podbod
				\item druhý podbod
			\end{itemize}
		\item Druhý bod
			\begin{enumerate}
				\item bod 1
			\end{enumerate}
		\item Třetí bod
			\begin{description}
				\item[Výzkum] \dots aneb vědecká honba za poznáním
			\end{description}
	\end{itemize}
\end{frame}

\subsection{Text}
\begin{frame}[nobg]
	\frametitle{Slide bez pozadí}
	Tento slide nemá žádné pozadí.
\end{frame}

\subsection{Bloky}
\begin{frame}
	\frametitle{Bloky}
	\begin{definition}[Pozdrav]
		Hello World!
	\end{definition}

	\begin{alertblock}{Chyba.}
		Něco zde nefunguje.
	\end{alertblock}

	\begin{exampleblock}{Správně!}
		Toto je lepší.
	\end{exampleblock}

\end{frame}

\begin{plainFrame}[colourbg]
	\plainFrameText{Děkuji za pozornost}
\end{plainFrame}

\end{frame}

\end{document}
